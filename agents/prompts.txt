async def search_column_values(
        self,
        search_terms: List[str],
        column_index: Dict[str, Any],
        max_columns: int = 7,
        max_values_per_column: int = 5
    ) -> List[str]:
        """
        Search for values across columns with tiered matching.
        
        Priority:
            Tier 1: Exact match (value == search term)
            Tier 2: Prefix match (value starts with search term)
            Tier 3: Contains ALL words (for multi-word search)
            Tier 4: Contains SOME words (fallback only if no Tier 1-3)
        """
        
        if not search_terms:
            return []
        
        columns_data = column_index.get("columns", {})
        
        if not columns_data:
            print("‚ö†Ô∏è No columns data found in column index")
            return []
        
        # Build search list
        search_phrases = []  # Full phrases
        
        for term in search_terms:
            term_lower = term.strip().lower()
            if term_lower:
                words = term_lower.split()
                search_phrases.append({
                    "phrase": term_lower,
                    "words": [w for w in words if len(w) > 2],
                    "is_multi_word": len(words) > 1
                })
        
        print(f"üîç Search phrases: {[s['phrase'] for s in search_phrases]}")
        
        # Track results per column
        # Structure: {column_name: {"tier": int, "values": {value: {"tier": int, "matched_words": int}}}}
        column_matches = {}
        
        for col_name, values in columns_data.items():
            
            for value in values:
                best_tier = None
                best_matched_words = 0
                
                for search_item in search_phrases:
                    phrase = search_item["phrase"]
                    words = search_item["words"]
                    is_multi_word = search_item["is_multi_word"]
                    
                    # TIER 1: Exact match
                    if value == phrase:
                        tier = 1
                        matched_words = len(words)
                    
                    # TIER 2: Prefix match
                    elif value.startswith(phrase):
                        tier = 2
                        matched_words = len(words)
                    
                    # TIER 3/4: Contains logic
                    else:
                        # Count how many words are in the value
                        words_found = [w for w in words if w in value]
                        matched_words = len(words_found)
                        
                        if matched_words == 0:
                            continue  # No match at all
                        
                        if is_multi_word:
                            if matched_words == len(words):
                                # ALL words found ‚Üí Tier 3
                                tier = 3
                            else:
                                # SOME words found ‚Üí Tier 4 (low priority)
                                tier = 4
                        else:
                            # Single word search, contains match ‚Üí Tier 3
                            tier = 3
                    
                    # Keep best tier (lowest number)
                    if best_tier is None or tier < best_tier:
                        best_tier = tier
                        best_matched_words = matched_words
                    elif tier == best_tier and matched_words > best_matched_words:
                        best_matched_words = matched_words
                
                # Record match if found
                if best_tier is not None:
                    if col_name not in column_matches:
                        column_matches[col_name] = {"best_tier": best_tier, "values": {}}
                    
                    if best_tier < column_matches[col_name]["best_tier"]:
                        column_matches[col_name]["best_tier"] = best_tier
                    
                    column_matches[col_name]["values"][value] = {
                        "tier": best_tier,
                        "matched_words": best_matched_words
                    }
        
        # CRITICAL: If Tier 1, 2, or 3 matches exist, remove Tier 4
        has_good_matches = any(
            col_data["best_tier"] <= 3 
            for col_data in column_matches.values()
        )
        
        if has_good_matches:
            for col_name, col_data in list(column_matches.items()):
                # Remove Tier 4 values
                col_data["values"] = {
                    v: info for v, info in col_data["values"].items()
                    if info["tier"] <= 3
                }
                
                # Recalculate best tier
                if col_data["values"]:
                    col_data["best_tier"] = min(info["tier"] for info in col_data["values"].values())
            
            # Remove columns with no values left
            column_matches = {k: v for k, v in column_matches.items() if v["values"]}
        
        # Sort columns: tier (asc), then column name
        sorted_columns = sorted(
            column_matches.items(),
            key=lambda x: (x[1]["best_tier"], x[0])
        )
        
        # Format output
        results = []
        for col_name, col_data in sorted_columns[:max_columns]:
            # Sort values: tier (asc), matched_words (desc)
            sorted_values = sorted(
                col_data["values"].items(),
                key=lambda x: (x[1]["tier"], -x[1]["matched_words"])
            )
            
            top_values = [v[0] for v in sorted_values[:max_values_per_column]]
            values_str = ", ".join(top_values)
            
            tier = col_data["best_tier"]
            results.append(f"{col_name} - {values_str}")
            print(f"   Tier {tier}: {col_name} ‚Üí {top_values}")
        
        print(f"‚úÖ Search complete: Found {len(results)} matching columns")
        return results
